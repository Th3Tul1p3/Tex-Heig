\section{Module de $Z$}
\begin{align*}
   Z &= a+jb \\
   |Z| &= \sqrt{a^2+b^2} & a,b ∈ ℝ
   |Z|^2 &= ZZ* \\
   |Z| = \sqrt{ZZ*} \\
\end{align*}
\begin{align*}
   |Z| &\geqslant 0 \\
   |Z| &= d(0,Z) \\
\end{align*}

\subsection{Propriétés}
\begin{align}
   |Z| &\geqslant 0 \\
   Z_1 &= a_1 + b_1j & Z_2 &= a_2 + b_2j & |Z_1Z_2| &= |Z_1||Z_2|  \\
   Z_1 &= a_1 + b_1j & Z_2 &= a_2 + b_2j & |\frac{Z_1}{Z_2}| &= \frac{|Z_1|}{|Z_2|} \\
   |Z|^2 &= ZZ* \\
   |Z| &= \sqrt{ZZ*} \\
\end{align}
\begin{align*}
   ZZ* &= (a+jb)(a-jb) = a^2-(jb)^2 = a^2 + b^2 &&= |Z|^2
\end{align*}
